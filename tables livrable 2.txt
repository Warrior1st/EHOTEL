CREATE TABLE chainehotel(
    chainehotel_id int
	nom VARCHAR(250),
	adresse_bureau VARCHAR(250),
	courriel VARCHAR (250),
	phone VARCHAR (250),
	nbr_hotel NUMERIC,
	PRIMARY KEY(chainehotel_id)	
);
CREATE TABLE commodite(
	commodite_id VARCHAR(250),
	nom VARCHAR(250),
	equipement VARCHAR(250),
	prix INTEGER ,
	PRIMARY KEY (commodite_id)
);
CREATE TABLE caracteristique (
	caracteristique_id VARCHAR(250),
	vue VARCHAR(250),
	extensibilite VARCHAR(250),
	dommage VARCHAR(250),
	PRIMARY KEY (caracteristique_id )
);
CREATE TABLE hotel(
	hotel_id VARCHAR(250),
	chainehotel_id VARCHAR(250),
	nbr_chambre NUMERIC,
	adresse VARCHAR(250),
	phone VARCHAR (250),
	courriel VARCHAR (250),
	classement INTEGER,
	PRIMARY KEY (hotel_id)
	
);

alter table hotel add foreign key references chainehotel (chainehotel_id);

CREATE TABLE reservation (
	reservation_id VARCHAR(250),
	client_id VARCHAR(250),
	chambre_id VARCHAR(250),
	PRIMARY KEY(reservation_id)
);
